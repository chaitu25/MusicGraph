What Grammy Award winning songs were also top spotify songs in the year 2016?
select distinct ?Song ?Artist (str(?year) as ?Year) where { ?award a music:GrammyAward; music:awardedFor ?Song; music:awardYear ?year; music:awardedTo ?artist; music:awardTitle ?GrammyAward. ?artist foaf:name ?Artist. Filter(str(?year)="2016"). {SELECT ?Song where{ ?track a mo:Track; music:title ?Song. }} }

What top Spotify songs in the year 2018 were made by artists from the United States?
SELECT distinct ?Track ?Artist (str(?year) as ?Year) ?Country WHERE { ?track a mo:Track; music:title ?Track; music:sungBy ?artist; music:year ?year. ?artist foaf:name ?Artist. ?artist music:country ?Country. Filter(?Year="2018" && ?Country="United States"). }
my(table2,query2)
Find the danceability of the song that won “Record Of The Year” in 2011.
SELECT distinct ?Song ?Artist ?Award (str(?year) as ?Year) ?danceability WHERE { ?award a music:GrammyAward; music:awardedFor ?Song; music:awardYear ?year; music:awardedTo ?artist; music:awardTitle ?Award. ?artist foaf:name ?Artist. filter(?Award='Record Of The Year' && str(?year)='2011') {SELECT ?Song ?danceability where{ ?track a mo:Track; music:title ?Song; music:dancability ?danceability. }} }

How many FM listeners does the artist who won “Best Pop Solo Performance” in 2017 have?
SELECT ?Grammy ?Artist (str(?year) as ?Year) ?Track ?FM_Listeners WHERE { ?awardType a music:GrammyAward; music:awardTitle ?Grammy; music:awardedTo ?artist; music:awardYear ?year; music:awardedFor ?Track. ?artist foaf:name ?Artist. ?artist music:listeners ?FM_Listeners Filter(?Grammy="Best Pop Solo Performance" && ?Year='2017'). }

Find the bpm of the song that won “Song of the Year” in the year 2016?
SELECT distinct ?Song ?Artist (str(?bpm) as ?BPM) ?Award (str(?year) as ?Year) WHERE { ?award a music:GrammyAward; music:awardedFor ?Song; music:awardYear ?year; music:awardedTo ?artist; music:awardTitle ?Award. ?artist foaf:name ?Artist. filter(?Award='Song Of The Year' && str(?year)='2016') {SELECT ?Song ?bpm where{ ?track a mo:Track; music:title ?Song; music:bpm ?bpm. }} } limit 1

Find 5 Grammy Award winning songs that have a bpm higher than 100.
SELECT distinct ?Song ?Artist (str(?bpm) as ?BPM) WHERE { ?award a music:GrammyAward; music:awardedFor ?Song; music:awardYear ?year; music:awardedTo ?artist; music:awardTitle ?Award. ?artist foaf:name ?Artist. {SELECT ?Song ?bpm where{ ?track a mo:Track; music:title ?Song; music:bpm ?bpm. filter(?bpm > 100) }} } order by asc(?bpm) limit 5
