SPARQL Queries

Query 1: What Grammy Award winning songs were also top spotify songs in the year 2016? 

PREFIX mo: <http://purl.org/ontology/mo/>
PREFIX dc: <http://purl.org/dc/elements/1.1/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX ex: <http://data.example.org>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>
PREFIX music: <http://www.music.org/ontologies/music#>

select distinct ?Song ?Artist (str(?year) as ?Year)
where {
	?award a music:GrammyAward;
		music:awardedFor ?Song;
		music:awardYear ?year;
		music:awardedTo ?artist;
		music:awardTitle ?GrammyAward.
	?artist foaf:name ?Artist.
	Filter(str(?year)="2016").
	{SELECT ?Song
	where{
		?track a mo:Track;
			music:title ?Song.
	}}
}

Query 2: What top Spotify songs in the year 2018 were made by artists from the United States? 

PREFIX mo: <http://purl.org/ontology/mo/>
PREFIX dc: <http://purl.org/dc/elements/1.1/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX ex: <http://data.example.org>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>
PREFIX music: <http://www.music.org/ontologies/music#>

SELECT distinct	?Track ?Artist (str(?year) as ?Year) ?Country
WHERE	{	
	?track a mo:Track;
			music:title ?Track;
			music:sungBy ?artist;
			music:year ?year.
	?artist foaf:name ?Artist.
	?artist music:country ?Country.
	Filter(?Year="2018" && ?Country="United States").
}

Query 7: Find 5 songs written by singers from the United States that have an energy higher than 70. 

PREFIX mo: <http://purl.org/ontology/mo/>
PREFIX dc: <http://purl.org/dc/elements/1.1/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX ex: <http://data.example.org>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>
PREFIX music: <http://www.music.org/ontologies/music#>

SELECT ?Singer ?Song ?Energy
WHERE	{	
	?singer a music:Singer;
			foaf:name ?Singer;
			music:country ?Country.
	filter(?Country='United States')
	{SELECT ?Energy ?Song ?singer
		WHERE {
			?song music:sungBy ?singer;
					music:energy ?Energy;
					music:title ?Song.
			filter(?Energy > 70)
		} 
	}
} limit 10

Query 8: Get the top songs of the artist with the most FM Listeners. 

PREFIX mo: <http://purl.org/ontology/mo/>
PREFIX dc: <http://purl.org/dc/elements/1.1/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX ex: <http://data.example.org>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>
PREFIX music: <http://www.music.org/ontologies/music#>
SELECT	?name ?Artist
WHERE	{
	?song ex:sungBy ?artist;
			dc:title ?name.
	{
		SELECT ?Artist 
		WHERE	{
			?artist ex:noOfListerns ?noOfListerns;
					foaf:name ?Artist.
		} ORDER BY DESC(xsd:integer(?noOfListerns)) LIMIT 1
	}
	?artist foaf:name ?Artist
}

Query 9: Find 10 songs with the same genre as the song that won “Best Pop Solo Performance” in 2016? 

PREFIX mo: <http://purl.org/ontology/mo/>
PREFIX dc: <http://purl.org/dc/elements/1.1/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX ex: <http://data.example.org>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>
PREFIX music: <http://www.music.org/ontologies/music#>

select distinct ?Song ?Genre (str(?year) as ?Year)
where {
	?grammy a music:GrammyAward;
			music:awardTitle ?Award;
			mo:Genre ?Genre;
			music:awardYear ?grammyYear.
			Filter(str(?grammyYear)="2016" && ?Award="Best Pop Solo Performance")
	bind (?Genre as ?GrammyGenre)
	{select ?Song ?Genre ?year
		where {
			?song a mo:Track;
					music:year ?year;
					music:title ?Song;
					mo:Genre ?GrammyGenre.
		}
	}
} limit 10

Query 10: What other artist/s released songs the same year that Kendrick Lamar won “Best Rap Song”? 

PREFIX mo: <http://purl.org/ontology/mo/>
PREFIX dc: <http://purl.org/dc/elements/1.1/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX ex: <http://data.example.org>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>
PREFIX music: <http://www.music.org/ontologies/music#>

SELECT distinct ?Artist (str(?year) as ?Year)
WHERE{
	?song mo:Genre ?genre;
		dc:title ?name;
		ex:sungBy ?artist;
		dc:date ?year.
	{
	select (str(?year) as ?Year_Exp)
		where {
			?award a mo:Award;
				ex:awardedTo 'Kendrick Lamar';
				dc:title 'Best Rap Song';
				dc:date ?year.
		}
	}
	FILTER(str(?year) = ?Year_Exp).
	?artist foaf:name ?Artist
}
